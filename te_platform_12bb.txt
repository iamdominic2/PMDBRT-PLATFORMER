//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Looks.SetSizeTo(550);
    Looks.Hide();
    Motion.GoToXY(0, -65);
    Control.Wait(0.33);
    Forever
    {
        Motion.TurnRight(3);
    }
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(ext)
{
    Control.Wait(0);
    Looks.Hide();
    If (Backdrop.GetNumberName(number) ==  60)
    {
        Looks.Show();
    }
}
