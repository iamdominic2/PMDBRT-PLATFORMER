//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Looks.Hide();
    Motion.GoToXY(16, 116);
    Control.Wait(0.33);
    Forever
    {
        Repeat (9)
        {
            Motion.ChangeYBy(11);
        }
        Repeat (9)
        {
            Motion.ChangeYBy(-11);
        }
    }
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(ext)
{
    Control.Wait(0);
    Looks.Hide();
    If (Backdrop.GetNumberName(number) == 59)
    {
        Looks.Show();
    }
}
