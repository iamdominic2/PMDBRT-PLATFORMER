//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Looks.Hide();
    Motion.GoToXY(-118, 0);
    Control.Wait(0.33);
    Forever
    {
        Motion.GlideTo(197, 100, 1.2);
        Motion.GlideTo(190, 100, 1.2);
    }
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(ext)
{
    Control.Wait(0);
    Looks.Hide();
    If (Backdrop.GetNumberName(number) == 51)
    {
        Looks.Show();
    }
}
