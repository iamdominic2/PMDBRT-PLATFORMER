//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Looks.Hide();
    Motion.GoToXY(-75, 50);
    Repeat (16)
    {
        Motion.ChangeYBy(-3);
    }
    Forever
    {
        Repeat (33)
        {
            Motion.ChangeYBy(-2);
        }
        Repeat (33)
        {
            Motion.ChangeYBy(2);
        }
    }
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(ext)
{
    Control.Wait(0);
    Looks.Hide();
    If (Backdrop.GetNumberName(number) == 28)
    {
        Looks.Show();
    }
}
