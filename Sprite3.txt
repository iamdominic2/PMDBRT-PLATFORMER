//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Looks.Hide();
    Motion.GoToXY(146, -55);
    Control.Wait(0.33);
    Forever
    {
        Repeat (13)
        {
            Motion.ChangeYBy(11);
        }
        Repeat (26)
        {
            Motion.ChangeYBy(-5.5);
        }
    }
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(ext)
{
    Control.Wait(0);
    Looks.Hide();
    If (Backdrop.GetNumberName(number) == 62)
    {
        Looks.Show();
    }
}
