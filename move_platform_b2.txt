//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Looks.Hide();
    Motion.GoToXY(0, 0);
    Control.Wait(0.33);
    Forever
    {
        Motion.GlideTo(-120, -120, 1);
        Control.Wait(0.5);
        Motion.GlideTo(130, -120, 1.5);
    }
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(ext)
{
    Control.Wait(0);
    Looks.Hide();
    If (Backdrop.GetNumberName(number) == 63)
    {
        Looks.Show();
    }
}
