//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Looks.Hide();
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(ext)
{
    Repeat (1)
    {
        Control.DeleteThisClone();
    }
    If (Backdrop.GetNumberName(number) == 49)
    {
        Forever
        {
            Motion.GoToXY(80, 111);
            Motion.PointInDirection(-90);
            Control.CreateCloneOf(_myself_);
            Control.Wait(1);
        }
    }
}


//----- Start as clone ------------------------------------------------------------------------------------------------

Control.WhenIStartAsClone()
{
    Looks.Show();
    Repeat Until Sensing.TouchingObject(_edge_)
    {
        Motion.Move(9);
    }
    Control.DeleteThisClone();
}


//----- Orphaned blocks -----------------------------------------------------------------------------------------------

Motion.PointInDirection(0);
Control.CreateCloneOf(_myself_);
Motion.PointInDirection(180);
Control.CreateCloneOf(_myself_);
Control.Wait(1);
