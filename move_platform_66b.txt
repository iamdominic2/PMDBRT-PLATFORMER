//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Looks.Hide();
    Motion.GoToXY(150, -122);
    Control.Wait(0.33);
    Forever
    {
        Motion.GlideTo(-118, -126, 1.4);
        Motion.GlideTo(150, -116, 1.5);
    }
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(ext)
{
    Control.Wait(0);
    Looks.Hide();
    If (Backdrop.GetNumberName(number) == 50)
    {
        Looks.Show();
    }
}
