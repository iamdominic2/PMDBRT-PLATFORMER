//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Looks.Hide();
    Motion.GoToXY(-8, -89);
    Control.Wait(0.33);
    Forever
    {
        Motion.GlideTo(166, -84, 1);
        Motion.GlideTo(-33, -89, 0.4);
    }
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(ext)
{
    Control.Wait(0);
    Looks.Hide();
    If (Backdrop.GetNumberName(number) == 45)
    {
        Looks.Show();
    }
}
