//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Looks.Hide();
    Motion.GoToXY(0, 50);
    Control.Wait(0.33);
    Forever
    {
        Repeat (20)
        {
            Motion.ChangeYBy(-5);
        }
        Repeat (20)
        {
            Motion.ChangeYBy(5);
        }
    }
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(ext)
{
    Control.Wait(0);
    Looks.Hide();
    If (Backdrop.GetNumberName(number) == 44)
    {
        Looks.Show();
    }
}
