//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Looks.Hide();
    Motion.GoToXY(-71, 134);
    Control.Wait(0.33);
    Forever
    {
        Repeat (21)
        {
            Motion.ChangeYBy(-11);
        }
        Repeat (21)
        {
            Motion.ChangeYBy(11);
        }
    }
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(ext)
{
    Control.Wait(0);
    Looks.Hide();
    If (Backdrop.GetNumberName(number) == 49)
    {
        Looks.Show();
    }
}
