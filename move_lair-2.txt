//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Looks.Hide();
    Motion.GoToXY(114, 0);
    Forever
    {
        Motion.GlideTo(133, 63, 2);
        Motion.GlideTo(-75, 64, 2);
    }
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(ext)
{
    Control.Wait(0);
    Looks.Hide();
    If (Backdrop.GetNumberName(number) == 43)
    {
        Looks.Show();
    }
}
