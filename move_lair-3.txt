//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Looks.Hide();
    Motion.GoToXY(114, 0);
    Forever
    {
        Motion.GlideTo(-114, 0, 1.5);
        Motion.GlideTo(0, 0, 1.5);
    }
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(ext)
{
    Control.Wait(0);
    Looks.Hide();
    If (Backdrop.GetNumberName(number) == 51)
    {
        Looks.Show();
    }
}
