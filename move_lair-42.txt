//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Looks.Hide();
    Motion.GoToXY(114, 0);
    Forever
    {
        Repeat (21)
        {
            Motion.ChangeYBy(-4);
        }
        Repeat (21)
        {
            Motion.ChangeYBy(4);
        }
    }
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(ext)
{
    Control.Wait(0);
    Looks.Hide();
    If (Backdrop.GetNumberName(number) == 42)
    {
        Looks.Show();
    }
}
