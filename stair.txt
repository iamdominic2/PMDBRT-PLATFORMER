//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Motion.GoToXY(210, -155);
    Looks.SetSizeTo(400);
    Call du7mp;
    Control.Wait(0.01);
    Call du7mp;
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(ext)
{
    Control.Wait(0.01);
    Motion.GoToXY(210, -240);
    Call du7mp;
}
