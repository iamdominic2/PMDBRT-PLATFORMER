//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Looks.Hide();
    Motion.GoToXY(-155, -11);
    Control.Wait(1);
    Forever
    {
        Repeat (55)
        {
            Motion.ChangeXBy(7);
        }
        Repeat (55)
        {
            Motion.ChangeXBy(-7);
        }
    }
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(ext)
{
    Control.Wait(0);
    Looks.Hide();
    If (Backdrop.GetNumberName(number) == 29)
    {
        Looks.Show();
    }
}
