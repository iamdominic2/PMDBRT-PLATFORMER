//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Looks.Hide();
    Motion.GoToXY(-200, -145);
    Control.Wait(0.33);
    Forever
    {
        Repeat (46)
        {
            Motion.ChangeXBy(7);
            Motion.ChangeYBy(5);
        }
        Repeat (46)
        {
            Motion.ChangeXBy(-7);
            Motion.ChangeYBy(-5);
        }
    }
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(ext)
{
    Control.Wait(0);
    Looks.Hide();
    If (Backdrop.GetNumberName(number) == 33)
    {
        Looks.Show();
    }
}
